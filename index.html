Animations

 Just add the classes "animate", and add the name of the animation! 
 <!-- Link to Feather -->
<link rel="stylesheet" type="text/css" href="https://raw.githubusercontent.com/KingPixil/Feather/gh-pages/feather.min.css"/>
<!-- Make a row with everything in the center -->
<div class="row text-center">
  <div class="pull-left animate bounce infinite">
    Feather
  </div>
<!-- Buttons => Add class btn + btn-(1-5) -->
<button class="btn btn-[1-5] ">Button</button>
<!-- Space => Put a type of space-->
<div class="small-space"></div>
<div class="space"></div>
<div class="xtra-small-space"></div>
<!-- Range => Add class range + blue-->
<input type="range" min="-100" max="0" value="0" class="range blue"/>
<!-- Checkbox -->
<input type="checkbox" name="checkbox" id="checkbox:checkbox-1"/>
    <label for="checkbox:checkbox-1" class="checkbox-label">Checkbox</label>
</div>
